# Custom Providers

Create custom providers to connect Open Harness to any AI model or service.

## Provider Interface

Every provider implements the `Provider` interface:

```typescript
import type { Provider, ProviderInput, Signal } from "@open-harness/core";

interface Provider {
  run(input: ProviderInput, context: RunContext): AsyncGenerator<Signal>;
}

interface RunContext {
  signal: AbortSignal;  // For cancellation
}
```

## Minimal Provider

A minimal provider that echoes input:

```typescript
import { createSignal, type Provider, type ProviderInput, type Signal } from "@open-harness/core";

class EchoProvider implements Provider {
  async *run(input: ProviderInput): AsyncGenerator<Signal> {
    // Emit start signal
    yield createSignal("provider:start", { model: "echo" });

    // Extract prompt from messages
    const prompt = input.messages
      .filter(m => m.role === "user")
      .map(m => m.content)
      .join("\n");

    // Emit text (simulate streaming)
    yield createSignal("text:delta", { content: `Echo: ${prompt}` });
    yield createSignal("text:complete", { content: `Echo: ${prompt}` });

    // Emit end signal with usage
    yield createSignal("provider:end", {
      usage: { inputTokens: prompt.length, outputTokens: prompt.length + 6 },
    });
  }
}
```

## Using Your Provider

```typescript
import { createHarness } from "@open-harness/core";

type State = { input: string; output: string | null };
const { agent, runReactive } = createHarness<State>();

const myAgent = agent({
  prompt: "{{ state.input }}",
  activateOn: ["harness:start"],
  updates: "output",
});

const result = await runReactive({
  agents: { myAgent },
  state: { input: "Hello", output: null },
  provider: new EchoProvider(),
  endWhen: (s) => s.output !== null,
});

console.log(result.state.output); // "Echo: Hello"
```

## Required Signals

Providers must emit these signals:

| Signal | When | Payload |
|--------|------|---------|
| `provider:start` | At start | `{ model: string }` |
| `provider:end` | At completion | `{ usage?: { inputTokens, outputTokens }, cost?, sessionId? }` |
| `provider:error` | On error | `{ error: string }` |

## Text Signals

For streaming text output:

```typescript
yield createSignal("text:delta", { content: "partial text" });
yield createSignal("text:complete", { content: "full text" });
```

## Tool Signals

For tool/function calling:

```typescript
yield createSignal("tool:call", {
  id: "call_123",
  name: "search",
  input: { query: "..." },
});

// After tool execution
yield createSignal("tool:result", {
  id: "call_123",
  name: "search",
  output: { results: [...] },
});
```

## OpenAI-Compatible Provider

Example wrapping an OpenAI-compatible API:

```typescript
import { createSignal, type Provider, type ProviderInput, type Signal } from "@open-harness/core";

class OpenAICompatibleProvider implements Provider {
  constructor(private config: { apiKey: string; baseUrl: string; model: string }) {}

  async *run(input: ProviderInput): AsyncGenerator<Signal> {
    yield createSignal("provider:start", { model: this.config.model });

    try {
      const response = await fetch(`${this.config.baseUrl}/chat/completions`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": `Bearer ${this.config.apiKey}`,
        },
        body: JSON.stringify({
          model: this.config.model,
          messages: input.messages,
          stream: true,
        }),
      });

      // Handle streaming response
      const reader = response.body?.getReader();
      const decoder = new TextDecoder();
      let fullContent = "";

      while (reader) {
        const { done, value } = await reader.read();
        if (done) break;

        const chunk = decoder.decode(value);
        // Parse SSE and extract content...
        const content = parseSSEChunk(chunk);
        if (content) {
          fullContent += content;
          yield createSignal("text:delta", { content });
        }
      }

      yield createSignal("text:complete", { content: fullContent });
      yield createSignal("provider:end", {
        usage: { inputTokens: 0, outputTokens: fullContent.length },
      });
    } catch (error) {
      yield createSignal("provider:error", { error: String(error) });
    }
  }
}
```

## Cancellation Support

Respect the abort signal for cancellation:

```typescript
async *run(input: ProviderInput, context: RunContext): AsyncGenerator<Signal> {
  yield createSignal("provider:start", { model: "my-model" });

  for (const chunk of someStreamingSource) {
    // Check for cancellation
    if (context.signal.aborted) {
      yield createSignal("provider:error", { error: "Cancelled" });
      return;
    }

    yield createSignal("text:delta", { content: chunk });
  }

  yield createSignal("provider:end", { usage: { inputTokens: 0, outputTokens: 0 } });
}
```

## Testing Your Provider

Test with the signal matchers:

```typescript
import { describe, it, expect } from "vitest";
import { createHarness } from "@open-harness/core";
import { toContainSignal, toHaveSignalsInOrder } from "@open-harness/vitest";

expect.extend({ toContainSignal, toHaveSignalsInOrder });

describe("MyProvider", () => {
  it("emits required signals", async () => {
    const { agent, runReactive } = createHarness<{ input: string; output: string | null }>();

    const testAgent = agent({
      prompt: "{{ state.input }}",
      activateOn: ["harness:start"],
      updates: "output",
    });

    const result = await runReactive({
      agents: { testAgent },
      state: { input: "test", output: null },
      provider: new MyProvider(),
      endWhen: (s) => s.output !== null,
    });

    expect(result.signals).toContainSignal("provider:start");
    expect(result.signals).toContainSignal("provider:end");
    expect(result.signals).toHaveSignalsInOrder([
      "provider:start",
      "text:complete",
      "provider:end",
    ]);
  });
});
```
